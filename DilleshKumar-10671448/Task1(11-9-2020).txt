function  jfun1()
                {
                    
                    var name = form1.name.value;
                    var designation = form1.designation.value;

                    if(name=="")
                    {
                        alert("Please Enter the Name");
                        form1.name.focus();
                        return false;
                        
                    }
                    
                    else if(designation=="")
                    {
                        alert("Please Enter the Designation");
                        form1.designation.focus();
                        return false;

                    }

                    else
                    {
                        return true;

                    }
                }
